# Функции в JavaScript

    // Объявление, определение функции
    function hello() {
        console.log('hello')
    }
    
    // Вызов функции
    hello()

## Параметры и аргументы функции
    // Параметры
    function hello(name, family) {
        console.log('Hello ' + name + ' ' + family)
    }
    
    // Аргументы
    hello('John', 'Smith')
