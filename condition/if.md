# if
Если условие `true` то остальные условия не выполняются.

    if (условие) {
        действие
    }

    if (условие) действие // сокращённая запись

## if ... else

    if (condition) {
        statement 1
    } else {
        statement 2
    }

## if ... else if ... else

    if (condition) {
        statement 1
    } else if (condition) {
        statement 2
    } else {
        statement 3
    }
